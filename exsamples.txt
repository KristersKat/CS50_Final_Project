    <a-scene>
      <!-- A button entity -->
      <a-entity
        a-gui-button="value: Click me; clickAction: myButtonClick"
        position="0 1 -3"
      ></a-entity>

      <!-- A text entity to display button click information -->
      <a-entity
        id="infoText"
        position="-1 0.8 -3"
        text="value: Click the button!"
      ></a-entity>
    </a-scene>

    <script>
      // Define the button click function
      function myButtonClick() {
        var infoText = document.getElementById("infoText");
        infoText.setAttribute("text", "value: Button clicked!");
      }
    </script>
